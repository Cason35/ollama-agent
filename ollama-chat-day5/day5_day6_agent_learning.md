### 🧩 任务拆解

#### Step 1：实现 fakeSearch 工具函数

``` ts
function fakeSearch(keyword: string) {
  const mockData: Record<string, string> = {
    "北京天气": "北京今天晴，18~26℃",
    "上海天气": "上海今天多云，20~28℃",
    "Agent 是什么": "Agent 是能够感知、决策并执行动作的系统。"
  };

  return mockData[keyword] || `没有找到关于「${keyword}」的结果`;
}
```

------------------------------------------------------------------------

#### Step 2：后端调用工具

``` ts
if (parsed.action === "search") {
  const result = fakeSearch(parsed.keyword);

  return Response.json({
    type: "search",
    keyword: parsed.keyword,
    result
  });
}
```

------------------------------------------------------------------------

#### Step 3：前端展示结果

``` ts
if (res.type === "search") {
  setMessages(prev => [
    ...prev,
    {
      role: "assistant",
      content: `🔍 搜索关键词：${res.keyword}\n📌 结果：${res.result}`
    }
  ]);
}
```
